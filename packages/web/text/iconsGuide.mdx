import { Flex, IconButton, Grid } from "theme-ui";
import { Icon } from "@makerdao/dai-ui-icons";
import copy from "copy-to-clipboard";

## How to use the icons package

1. Install package <code>yarn add @makerdao/dai-ui-icons</code>
2. Import the icons into your theme object, and add to the theme provider.
3. Import the Icon component and use the name prop to reference your icon.

**Tip:** Click an icon to copy it's name, or click the button below to copy the code sample.

<Grid columns={["50px auto"]}>
  <IconButton sx={{ margin: "auto" }} onClick={() => copy(props.activeIcon)}>

      size="auto"
